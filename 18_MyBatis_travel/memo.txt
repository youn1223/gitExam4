drop table travel;
create table travel(
	num number primary key,
	name varchar2(10),
	age number,
	area varchar2(30),
	style varchar2(15),
	price varchar2(10)
);

drop sequence travel_seq;
create sequence travel_seq 
minvalue 1
start with 1
nocache;


insert into travel
values(travel_seq.nextval, '수지',21, '동남아','패키지','200~300');

commit;

col num for 99
col age for 999
col name for a6
col area for a25
col style for a10
col price for a10
